def get_week(weekday):

        return 'Weekday'
    else:
        return 'Weekend'

#function to generate categorical variable for season
def get_season(month):
    if month>2 and month<6:
        return 'Spring'
    elif month>5 and month<9:
        return 'Summer'
    elif month>8 and month<12:
        return 'Autumn'
    else:
        return 'Winter'
